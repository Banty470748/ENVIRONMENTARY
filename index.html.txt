    <script>
        // Simple State Management
        let trees = 0;
        const co2PerTree = 22; // kg per year
        const o2PerTree = 0.11; // tons (approx)

        function showLogin() {
            document.getElementById('login-overlay').style.display = 'flex';
        }

        function hideLogin() {
            document.getElementById('login-overlay').style.display = 'none';
        }

        function login() {
            // Simulate Login
            hideLogin();
            document.getElementById('landing-page').classList.add('hidden');
            document.getElementById('loginBtn').classList.add('hidden');
            document.getElementById('dashboard').style.display = 'block';
            document.getElementById('user-profile').classList.remove('hidden');
            
            // Simulate loading data
            setTimeout(() => {
                trees = 12; // Load "previous" data
                updateStats();
            }, 500);
        }

        function addTree() {
            trees++;
            // Visual feedback
            const btn = document.querySelector('.btn-action');
            const originalText = btn.innerText;
            btn.innerText = "✅ Logged!";
            btn.style.background = "#27ae60";
            
            setTimeout(() => {
                btn.innerText = originalText;
                btn.style.background = "#34495e";
            }, 1000);
            
            updateStats();
        }

        function updateStats() {
            // Animate numbers
            document.getElementById('tree-count').innerText = trees;
            document.getElementById('co2-count').innerText = (trees * co2PerTree).toFixed(1);
            document.getElementById('o2-count').innerText = (trees * o2PerTree).toFixed(2);
        }
    </script>
